## To greet the user, based on when the interaction time is.
def get_greeting(name):
 current_time = datetime.now().hour
 if 5 <= current_time < 12:
 return f"Good morning, {name}! How can I assist you today?"
 elif 12 <= current_time < 18:
 return f"Good afternoon, {name}! How can I assist you this afternoon?"
 else:
 return f"Good evening, {name}! How can I assist you tonight?"
def match_input(pattern, user_input):
 return bool(re.search(pattern, user_input, re.IGNORECASE))
